var tupleSameProduct = function(nums) {
    let ans = 0;
    const map = new Map();
    for(let i = 0; i < nums.length; i++) {
        for(let j = i + 1; j < nums.length; j++) {
            let key = nums[i] * nums[j]
            const count = map.get(key);
            if(!count) {
                map.set(key, 1);
            }else {
                map.set(key, count + 1);
            }
        }
    }
    for([key, value] of map) {
        ans +=(value * (value - 1)) / 2 * 8;
    }
    return ans;
};
