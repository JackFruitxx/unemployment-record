## 3. 每日一题

    1155. 掷骰子等于目标和的方法数


https://leetcode.cn/problems/number-of-dice-rolls-with-target-sum/

- 动态规划 n组背包

```javascript

var numRollsToTarget = function(n, k, target) {
    const mod = 1e9 + 7;
    f = new Array(target + 1).fill(0);
    f[0] = 1;
    for (let i = 1; i <= n; i++) {
        for (let j = target; j >= 0; j--) {
            f[j] = 0;
            for (let x = 1; x <= k; x++) {
                if (j - x >= 0) {
                    f[j] = (f[j] + f[j - x]) % mod;
                }
            }
        }
    }
    return f[target];
};
